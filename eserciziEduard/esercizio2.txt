--1) Elenco dei nomi dei corsi del tutor con indice 1


SELECT courses.name 

FROM courses

INNER JOIN tutorsassignment ON courses.id = tutorsassignment.coursee_id

WHERE tutorsassignment.tutor_id = 1;


--2)Elenco di righe aventi come colonne nome e cognome di un tutor ed uno dei corsi
--a cui è assegnato. (se un tutor lavora in più corsi, deve avere più righe corrispondenti)


SELECT tutors.first_name, tutors.last_name, courses.name

FROM tutors

INNER JOIN tutorsassignment ON tutors.id = tutorsassignment.tutor_id
INNER JOIN courses ON courses.id = tutorsassignment.coursee_id;

--3) Elenco dei tutor che non lavorano in nessun corso

SELECT * FROM tutors

WHERE id NOT IN (

    SELECT tutor_id
    FROM tutorsassignment

);


--4) Elenco dei tutor che hanno più di 5 anni di esperienza e lavorano in un solo corso, 
--riportando anche il nome del corso


SELECT tutors.first_name, tutors.last_name, courses.name AS course_name FROM tutors

JOIN tutorsassignment ON tutors.id = tutorsassignment.tutor_id
JOIN courses ON courses.id = tutorsassignment.coursee_id

WHERE tutors.exp_years > 5 AND (

    SELECT COUNT(*)
    FROM tutorsassignment
    WHERE tutor_id = tutors.id

) = 1;


--5) Elenco dei tutor che lavorano in un corso diverso dalla loro area di competenza, 
--riportando sia l'area di competenza che il nome del corso.


SELECT tutors.first_name, tutors.last_name, courses.name AS nome_corso, tutors.class FROM tutors

JOIN tutorsassignment ON tutors.id = tutorsassignment.tutor_id
JOIN courses ON courses.id = tutorsassignment.coursee_id

WHERE tutors.class != courses.name;
